--[[
neovim settings dependent on environment
]]

local M = {}

-- Provider settings
vim.g.python3_host_prog = "/usr/bin/python3"
vim.g.loaded_ruby_provider = 0
vim.g.loaded_perl_provider = 0

-- Visual Settings
vim.o.termguicolors = true
vim.o.guifont = "UDEV Gothic 35NF:h13"

-- Use extra settings
M.use_extra = false

-- Languages enabled in this environment.
M.enabled_languages = {
  rust = false,
  tex = false,
}

-- PDF viewer
-- This setting is used for synctex.
-- Sample for WSL with SmatraPDF
-- Must make ~/.local/bin/sumatrapdf.sh.
--[[ let g:vimtex_view_general_viewer = '~/.local/bin/sumatrapdf.sh'
  let g:vimtex_view_general_options = '-reuse-instance -forward-search @tex @line @pdf' ]]
M.pdf_viewer = "skim"
M.pdf_viewer_options = nil

-- LLM settings
M.llm = {
  enabled = false,
  end_point = "https://opencode.ai/zen/go/v1/chat/completions",
  provider_name = "OpenCode",
  api_key_env = "OPENCODE_API_KEY",
  model = "deepseek-v4-flash",
}

-- Markdown preview plugin selection
M.md_preview = {
  options = {
    MD_PREVIEW = "md-preview",
    PREVIEW = "preview",
  },
  prefer = "preview",
}

-- Returns the plugin to use based on environment
-- md-preview requires unix + glow, otherwise falls back to preview
function M.get_md_preview_plugin()
  if M.md_preview.prefer == M.md_preview.options.MD_PREVIEW then
    if vim.fn.has("unix") == 1 and vim.fn.executable("glow") == 1 then
      return M.md_preview.options.MD_PREVIEW
    else
      return M.md_preview.options.PREVIEW
    end
  end
  return M.md_preview.prefer
end

return M
