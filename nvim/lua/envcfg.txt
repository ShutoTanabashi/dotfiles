--[[
neovim settings dependent on environment
]]

local M = {}

-- Use extra settings
M.use_extra = false

-- PDF viewer
-- This setting is used for synctex.
M.pdf_viewer = "skim"
M.pdf_viewer_options = nil

-- Provider settings
vim.g.python3_host_prog = "/usr/bin/python3"
vim.g.loaded_ruby_provider = 0
vim.g.loaded_perl_provider = 0

-- Visual Settings
vim.o.termguicolors = true
vim.o.guifont = "UDEV Gothic 35NF:h13"

vim.g.use_extra = M.use_extra
if M.use_extra then
  vim.g.pdf_viewer = M.pdf_viewer
  vim.g.pdf_viewer_options = M.pdf_viewer_options
end

-- Sample for SumatraPDF
--[[ M.pdf_viewer = "SumatraPDF"
  M.pdf_viewer_options = '-reuse-instance -forward-search @tex @line @pdf' ]]

  let g:vimtex_view_general_options = '-reuse-instance -forward-search @tex @line @pdf' ]]

return M
